# Prime Contract Details API

The Prime Contract Details API allows you to retrieve detailed information about a specific prime contract for a project. This endpoint provides comprehensive data about the contract terms, values, and related information. {{ className: 'lead' }}

## Get Prime Contract Details

<CodeGroup tag="GET" label="/api/external/prime-contract-details">

```bash
curl -X GET "https://api.glapi.com/api/external/prime-contract-details?project_id=123&prime_contract_id=456" \
  -H "x-api-key: your-api-key" \
  -H "x-organization-id: your-organization-id" \
  -H "x-company-id: your-company-id"
```

</CodeGroup>

### Query Parameters

<Properties>
  <Property name="project_id" type="integer" required>
    The ID of the project that the prime contract belongs to.
  </Property>
  <Property name="prime_contract_id" type="integer" required>
    The ID of the specific prime contract to retrieve details for.
  </Property>
</Properties>

### Required Headers

<Properties>
  <Property name="x-api-key" type="string" required>
    Your API key for authentication.
  </Property>
  <Property name="x-organization-id" type="string" required>
    The ID of your organization.
  </Property>
  <Property name="x-company-id" type="string" required>
    The ID of the company you're accessing.
  </Property>
</Properties>

### Response

<ResponseField name="data" type="array">
  An array containing the prime contract details object.

  <Expandable title="Prime Contract Details Object">
    <ResponseField name="id" type="integer">
      The unique identifier for the prime contract.
    </ResponseField>
    <ResponseField name="title" type="string">
      The title of the prime contract.
    </ResponseField>
    <ResponseField name="description" type="string">
      A description of the prime contract.
    </ResponseField>
    <ResponseField name="contract_number" type="string">
      The contract number or identifier.
    </ResponseField>
    <ResponseField name="status" type="string">
      The current status of the contract (e.g., "Active", "Completed", "Terminated").
    </ResponseField>
    <ResponseField name="contract_date" type="string">
      The date when the contract was signed, in ISO 8601 format.
    </ResponseField>
    <ResponseField name="start_date" type="string">
      The start date of the contract, in ISO 8601 format.
    </ResponseField>
    <ResponseField name="estimated_completion_date" type="string">
      The estimated completion date of the contract, in ISO 8601 format.
    </ResponseField>
    <ResponseField name="original_contract_value" type="number">
      The original value of the contract.
    </ResponseField>
    <ResponseField name="current_contract_value" type="number">
      The current value of the contract, including any approved changes.
    </ResponseField>
    <ResponseField name="executed_change_orders_value" type="number">
      The total value of all executed change orders.
    </ResponseField>
    <ResponseField name="pending_change_orders_value" type="number">
      The total value of all pending change orders.
    </ResponseField>
    <ResponseField name="project_id" type="integer">
      The ID of the project that this contract belongs to.
    </ResponseField>
    <ResponseField name="customer" type="object">
      Information about the customer or client.
      <Expandable title="Customer Object">
        <ResponseField name="id" type="integer">
          The unique identifier for the customer.
        </ResponseField>
        <ResponseField name="name" type="string">
          The name of the customer.
        </ResponseField>
        <ResponseField name="contact_name" type="string">
          The name of the primary contact person.
        </ResponseField>
        <ResponseField name="contact_email" type="string">
          The email address of the primary contact person.
        </ResponseField>
        <ResponseField name="contact_phone" type="string">
          The phone number of the primary contact person.
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="payment_terms" type="object">
      Information about the payment terms of the contract.
      <Expandable title="Payment Terms Object">
        <ResponseField name="type" type="string">
          The type of payment terms (e.g., "Fixed Price", "Time and Materials").
        </ResponseField>
        <ResponseField name="payment_schedule" type="string">
          The payment schedule description (e.g., "Monthly", "Upon Completion").
        </ResponseField>
        <ResponseField name="retention_percentage" type="number">
          The retention percentage applied to payments.
        </ResponseField>
        <ResponseField name="days_until_due" type="integer">
          The number of days until payment is due after invoice.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
